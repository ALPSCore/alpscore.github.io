150  : in_(in)

152  EIGEN_STATIC_ASSERT_VECTOR_ONLY(Eigen::DenseBase<Derived>);
153  static_assert(std::is_same<T, typename Derived::Scalar>::value,
154  "Type mismatch -- use explicit cast");

162  throw size_mismatch();
