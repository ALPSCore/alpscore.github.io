118  communicator(const MPI_Comm& comm, comm_create_kind kind) {
119  switch (kind) {
120  default:
121  throw std::logic_error("alps::mpi::communicator(): unsupported `kind` argument.");
122  break;
123  case comm_attach:
124  comm_ptr_.reset(new MPI_Comm(comm));
125  break;
126  case take_ownership:
127  comm_ptr_.reset(new MPI_Comm(comm), comm_deleter());
128  break;
129  case comm_duplicate:
130  MPI_Comm* newcomm_ptr=new MPI_Comm();
131  MPI_Comm_dup(comm, newcomm_ptr);
132  comm_ptr_.reset(newcomm_ptr, comm_deleter());
133  break;
134  }
135  }

260  inline void broadcast(const communicator& comm, std::string& val, int root) {
261  std::size_t root_sz=val.size();
262  broadcast(comm, root_sz, root);
263  if (comm.rank()==root) {
264  // NOTE: at root rank the value being broadcast does not change, so const cast is safe
265  broadcast(comm, const_cast<char*>(val.data()), root_sz, root);
266  } else {
267  // FIXME: not very efficient --- any better way without heap alloc?
268  // Note, there is no guarantee in C++03 that modifying *(&val[0]+i) is safe!
269  std::unique_ptr<char[]> buf(new char[root_sz]);
270  broadcast(comm, buf.get(), root_sz, root);
271  val.assign(buf.get(), root_sz);
272  }
